<script>

let users = JSON.parse(localStorage.getItem("users")) || [];
let currentUser = null;
let pinInput = "";

// PIN
function addPin(n) {
    if(pinInput.length < 4) {
        pinInput += n;
        document.getElementById("pinDisplay").innerText = "*".repeat(pinInput.length);
    }
}

function clearPin() {
    pinInput = "";
    document.getElementById("pinDisplay").innerText = "****";
}

function checkPin() {
    if(pinInput === "0000") {
        document.getElementById("pinBox").style.display = "none";
        document.getElementById("loginBox").style.display = "block";
        document.getElementById("registerBox").style.display = "block";
    } else {
        alert("PIN incorrect");
        clearPin();
    }
}

// REGISTER
function register() {
    let name = newUser.value;
    let pass = newPass.value;

    users.push({name, pass, solde: 2000, historique: []});
    localStorage.setItem("users", JSON.stringify(users));
    alert("Compte créé");
}

// LOGIN
function login() {
    let name = user.value;
    let pass = pass.value;

    let u = users.find(x => x.name === name && x.pass === pass);

    if(u) {
        currentUser = u;
        loginBox.style.display = "none";
        registerBox.style.display = "none";
        compteBox.style.display = "block";
        update();
    } else {
        alert("Erreur");
    }
}

// UPDATE
function update() {
    solde.innerText = currentUser.solde;
    cardName.innerText = currentUser.name;

    let h = "";
    currentUser.historique.forEach(e => h += "<p>"+e+"</p>");
    historique.innerHTML = h;

    localStorage.setItem("users", JSON.stringify(users));
}

// ACTIONS
function deposer() {
    let m = parseInt(montant.value);
    currentUser.solde += m;
    currentUser.historique.push("Dépôt +"+m+"€");
    update();
}

function retirer() {
    let m = parseInt(montant.value);
    currentUser.solde -= m;
    currentUser.historique.push("Retrait -"+m+"€");
    update();
}

// TRANSFERT
function transfer() {
    let d = dest.value;
    let m = parseInt(montant.value);

    let receiver = users.find(u => u.name === d);

    if(receiver && m > 0) {
        currentUser.solde -= m;
        receiver.solde += m;

        currentUser.historique.push("Envoyé à "+d+" : -"+m+"€");
        receiver.historique.push("Reçu de "+currentUser.name+" : +"+m+"€");

        update();
    } else {
        alert("Erreur transfert");
    }
}

// LOGOUT
function logout() {
    location.reload();
}
